// ─── 🛡️ INPUT SANITIZATION — XSS Protection ──────────────────────────────────
function sanitizeInput(text, maxLength = 200) {
  if (!text || typeof text !== "string") return "";
  // HTML tags hataao (XSS prevention)
  let clean = text
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#x27;")
    .replace(/\//g, "&#x2F;");
  // Length limit
  return clean.slice(0, maxLength).trim();
}

// ─── 🔐 JWT TOKEN MANAGEMENT ──────────────────────────────────────────────────
// Token memory mein rakho — localStorage nahi (XSS safe)
let _jwtToken   = null;
let _csrfToken  = null;
let _sessionTimer = null;

function setAuthTokens(jwt, csrf) {
  _jwtToken  = jwt;
  _csrfToken = csrf;
}

function clearAuthTokens() {
  _jwtToken  = null;
  _csrfToken = null;
  if (_sessionTimer) clearTimeout(_sessionTimer);
}

function getAuthHeaders() {
  const headers = { "Content-Type": "application/json" };
  if (_jwtToken) headers["Authorization"] = `Bearer ${_jwtToken}`;
  return headers;
}

// ─── 🔐 API HELPERS — JWT + CSRF included ─────────────────────────────────────
async function apiLogin(username, password) {
  try {
    const res = await fetch(`${FLASK_BASE}/login`, {
      method: "POST",
      credentials: "include",
      headers: {
        "Content-Type": "application/json",
        "ngrok-skip-browser-warning": "true"   // ← ngrok warning page bypass
      },
      body: JSON.stringify({
        username: sanitizeInput(username, 50),
        password: password.slice(0, 100)
      })
    });
    const data = await res.json();
    if (data.status === "success") {
      // 🔐 Store JWT + CSRF tokens
      setAuthTokens(data.token, data.csrf_token);
    }
    return data;
  } catch {
    // Demo fallback (NO passwords exposed — just role info)
    // ❌ REMOVED: hardcoded password fallback
    return { status: "fail", message: "Cannot connect to server. Make sure Flask is running." };
  }
}

// 🚫 CSRF-protected POST helper
async function apiPost(endpoint, body = {}) {
  // Attach CSRF token to every state-changing request
  const payload = { ...body };
  if (_csrfToken) payload.csrf_token = _csrfToken;

  const res = await fetch(`${FLASK_BASE}${endpoint}`, {
    method: "POST",
    credentials: "include",
    headers: { ...getAuthHeaders(), "ngrok-skip-browser-warning": "true" },
    body: JSON.stringify(payload)
  });

  // If 401 → session expired
  if (res.status === 401) {
    clearAuthTokens();
    window.location.reload();
    return { status: "denied" };
  }

  // Refresh CSRF token after each use
  if (_jwtToken) {
    fetch(`${FLASK_BASE}/csrf_token`, { headers: getAuthHeaders() })
      .then(r => r.json())
      .then(d => { if (d.csrf_token) _csrfToken = d.csrf_token; })
      .catch(() => {});
  }

  return res.json();
}

  // ⏱️ Client-side brute force protection
  const [attempts, setAttempts] = useState(0);
  const [lockedUntil, setLockedUntil] = useState(null);
  const [timeLeft, setTimeLeft] = useState(0);

  // Countdown timer for lockout
  useEffect(() => {
    if (!lockedUntil) return;
    const interval = setInterval(() => {
      const left = Math.max(0, Math.ceil((lockedUntil - Date.now()) / 1000));
      setTimeLeft(left);
      if (left === 0) { setLockedUntil(null); setAttempts(0); }
    }, 1000);
    return () => clearInterval(interval);
  }, [lockedUntil]);

  const isLocked = lockedUntil && Date.now() < lockedUntil;

  const submit = async () => {
    // ⏱️ Check client-side lockout
    if (isLocked) {
      setErr(`Too many attempts. Wait ${timeLeft} seconds.`);
      return;
    }
    // 🛡️ Validation
    if (!u || !p) { setErr("Enter credentials."); return; }
    if (u.length > 50 || p.length > 100) { setErr("Invalid input length."); return; }
    // Basic username format check (no special chars)

    setLoading(true); setErr("");
    const res = await apiLogin(u, p);
    setLoading(false);

    if (res.status === "success") {
      onLogin({ username: res.username || u, role: res.role, area: res.area });

      // ⏱️ Track failed attempts
      const newAttempts = attempts + 1;
      setAttempts(newAttempts);
      if (newAttempts >= 5) {
        // Lock for 2 minutes client-side
        setLockedUntil(Date.now() + 2 * 60 * 1000);
        setErr("Too many failed attempts. Locked for 2 minutes.");
      } else {
        setErr(res.message || `Invalid credentials. ${5 - newAttempts} attempts remaining.`);

  const [toast, setToast]         = useState(null);
  const [showAddForm, setShowAddForm] = useState(false);
  const [resetTarget, setResetTarget] = useState(null);
  const [newPass, setNewPass]     = useState("");

  // New user form state
  const [newUser, setNewUser] = useState({
    username: "", password: "", role: "Inspector", assigned_area: ""
  });

  const LA_AREAS_LIST = [
    "All","Hollywood","Central","Newton","Southwest","Rampart",
    "77Th Street","Pacific","Olympic","Northeast","Van Nuys",
    "West La","Wilshire","N Hollywood","Southeast","Hollenbeck",
    "Harbor","Foothill","Mission","Devonshire","West Valley","Topanga"
  ];

  const roleColors = {
    Admin: "#f59e0b", SHO: "#3b82f6",
    Inspector: "#22c55e", User: "#94a3b8"
  };

  useEffect(() => { loadUsers(); }, []);

  const loadUsers = async () => {
    setLoading(true);
    const data = await apiGetUsers();
    setUsers(data.users || []);
    setLoading(false);
  };

  const showToast = (msg, type="success") => {
    setToast({ msg, type });
    setTimeout(() => setToast(null), 3000);
  };

  const handleAddUser = async () => {
    if (!newUser.username || !newUser.password) {
      showToast("Username aur Password required hai!", "error"); return;
    }
    if (newUser.password.length < 6) {
      showToast("Password kam se kam 6 characters ka hona chahiye!", "error"); return;
    }
    if (!/^[a-zA-Z0-9_]+$/.test(newUser.username)) {
      showToast("Username mein sirf letters, numbers aur _ allowed hain!", "error"); return;
    }
    const res = await apiAddUser({
      username: sanitizeInput(newUser.username, 50),
      password: newUser.password,
      role: newUser.role,
      assigned_area: sanitizeInput(newUser.assigned_area, 50)
    });
    if (res.status === "success") {
      showToast(`✅ ${newUser.username} successfully add ho gaya!`);
      setNewUser({ username:"", password:"", role:"Inspector", assigned_area:"" });
      setShowAddForm(false);
      loadUsers();

      showToast(res.message || "Error! Try again.", "error");
    }

    if (!window.confirm(`${uname} ko delete karna chahte ho?`)) return;
    const res = await apiDeleteUser(uid);
    if (res.status === "success") {
      showToast(`🗑️ ${uname} delete ho gaya!`);
      loadUsers();
    } else {
      showToast(res.message || "Delete failed!", "error");

      showToast(res.message || "Reset failed!", "error");
    }
  };

  const handleUnlock = async (uid, uname) => {
    const res = await apiUnlockUser(uid);
    if (res.status === "success") {
      showToast(`🔓 ${uname} ka account unlock ho gaya!`);
      loadUsers();
    } else {
      showToast("Unlock failed!", "error");
    }
  };

    color, borderRadius:8, cursor:"pointer", fontSize:11, fontWeight:600
  });

        <div>
          <div style={{ color:"white", fontWeight:600, fontSize:15 }}>👥 User Management</div>
          <div style={{ color:"#475569", fontSize:12, marginTop:4 }}>
            Admin can add, delete, reset passwords for all users
          </div>
        </div>
        <button onClick={() => setShowAddForm(!showAddForm)} style={{
          padding:"10px 20px", background:"linear-gradient(135deg,#16a34a,#22c55e)",
          border:"none", color:"white", borderRadius:10, cursor:"pointer",
          fontSize:12, fontWeight:700, letterSpacing:1
        }}>
          {showAddForm ? "✕ CANCEL" : "+ ADD USER"}
        </button>
      </div>

      {showAddForm && (
        <div style={{ background:"rgba(10,16,30,0.9)", border:"1px solid rgba(34,197,94,0.3)",

// ─── ROOT APP ─────────────────────────────────────────────────────────────────
export default function App() {
  const [session, setSession]     = useState(null);
  const [activeTab, setActiveTab] = useState("overview");
  const [collapsed, setCollapsed] = useState(false);
  const [sessionTimeLeft, setSessionTimeLeft] = useState(SESSION_TIMEOUT_MS);

  // ⏱️ Session timeout countdown
  useEffect(() => {
    if (!session) return;
    const start = Date.now();
    const interval = setInterval(() => {
      const elapsed = Date.now() - start;
      const left    = Math.max(0, SESSION_TIMEOUT_MS - elapsed);
      setSessionTimeLeft(left);
      if (left === 0) {
        // Auto logout on session expiry
        handleLogout();
      }
    }, 30000); // Check every 30 seconds
    return () => clearInterval(interval);
  }, [session]);

  const handleLogin = (s) => {
    setSession(s);
    setSessionTimeLeft(SESSION_TIMEOUT_MS);
    setActiveTab(s.role==="User" ? "chat" : s.role==="SHO" ? "cases" : "overview");
  };

  const handleLogout = async () => {
    try { await fetch(`${FLASK_BASE}/logout`, { credentials:"include", headers: getAuthHeaders() }); } catch {}
    clearAuthTokens();  // 🔐 Clear JWT + CSRF tokens
    setSession(null);
  };

  if (!session) return <LoginPage onLogin={handleLogin} />;

  return (
    <div style={{
      height:"100vh", display:"flex", flexDirection:"column",
      background:"#070b14", fontFamily:"'Courier New',monospace", overflow:"hidden"
    }}>
      <style>{`
        * { scrollbar-width: thin; scrollbar-color: #1e293b #070b14; }
        ::-webkit-scrollbar { width: 5px; }
        ::-webkit-scrollbar-track { background: #070b14; }
        ::-webkit-scrollbar-thumb { background: #1e293b; border-radius: 4px; }
        @keyframes slideIn { from{transform:translateX(30px);opacity:0} to{transform:none;opacity:1} }
        @keyframes pulse { 0%,100%{opacity:1} 50%{opacity:0.4} }
      `}</style>

      <Topbar role={session.role} user={session.username} activeTab={activeTab} sessionTimeLeft={sessionTimeLeft} />

      <div style={{ display:"flex", flex:1, overflow:"hidden" }}>
        <Sidebar
          role={session.role} active={activeTab} setActive={setActiveTab}
          user={session.username} area={session.area}
          onLogout={handleLogout} collapsed={collapsed} setCollapsed={setCollapsed}
        />
        <main style={{ flex:1, overflowY:"auto", padding:24 }}>
          {session.role==="Admin" && <AdminDashboard user={session.username} />}
          {session.role==="SHO"   && <SHODashboard   user={session.username} area={session.area} />}
          {session.role==="User"  && <PublicDashboard />}
        </main>
      </div>
    </div>
  );
}
